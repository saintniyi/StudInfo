<script src="~/lib/jquery/dist/jquery.min.js"></script>
<script src="//cdnjs.cloudflare.com/ajax/libs/toastr.js/latest/toastr.min.js"></script>

<script>


    $(document).ready(function() {
        addNewRow();
    })


    function addNewRow() {

        var row = `
                                <tr>
                                    <td><input type="text" class="form-control" id="txtGroupCode" name="txtGroupCode" /></td>
                                    <td><select id="ddlCourseCode" name="ddlCourseCode" class="form-select"></select></td>
                                    <td><select id="ddlStudNos" name="ddlStudNos" class="form-select"></select></td>
                                    <td><select id="ddlLevel" name="ddlLevel" class="form-select"></select></td>
                                    <td><select id="ddlSemester" name="ddlSemester" class="form-select"></select></td>
                                </tr>
                            `

        $("#myTable tbody").append(row);


        AssignValueToGroupCode()
        LoadCourseDropdown()
        LoadStudDropdown()
        LoadLevelDropdown()
        LoadSemesterDropdown()
    }


    function AssignValueToGroupCode() {
        var id = '@(ViewBag.id)' 

        $('#myTable tbody tr').each(function(row, ele) {

            let txtGroupCode = $(this).find('input[id=txtGroupCode]')
            txtGroupCode.val(id)
            txtGroupCode.prop('disabled', true);
        });

    }



    function LoadCourseDropdown() {

        $('#myTable tbody tr').each(function(row, ele) {

            let ddlCourseCode = $(this).find('select[id=ddlCourseCode]')

            /* these 2 lines of code prevent dropdown values in previous row(s) from loosing their values when a
            new row is added i.e. prevents selected item from resetting to the first item after a new row is added */
            kant = $('#myTable tbody tr').length
            if (row < (kant - 1)) return true

            $.ajax({
                type: "POST",
                url: "/Enrollment/LoadCourses",
                success: function(data) {
                    ddlCourseCode.empty()
                    $.each(data, function() {
                        //ddlCourseCode.append($("<option></option>").val(this['value']).html(this['text']));
                        ddlCourseCode.append($("<option />").text(this.value).text(this.text));
                    });

                }
            });

        })
    }

        $('#myTable tbody tr').each(function(row, ele) {

                    });

            });

        })
    }

            let ddlLevel = $(this).find('select[id=ddlLevel]')

    function Save() {

        enrollments = [];

        $('#myTable tbody tr').each(function(row, ele) {

            var enrollment = {}

            let gcode = $(this).find('input[name=txtGroupCode]').val()
            enrollment.GroupCode = gcode

            let ccode = $(this).find('select[name=ddlCourseCode]').val()
            enrollment.CourseCode = ccode

            let studnos = $(this).find('select[name=ddlStudNos]').val()
            enrollment.StudNos = studnos

            let level = $(this).find('select[name=ddlLevel]').val()
            enrollment.Level = level

            let semester = $(this).find('select[name=ddlSemester]').val()
            enrollment.Semester = semester

            if (ccode == '--Select--' || studnos == '--Select--' || level == '--Select--' || semester == '--Select--'){
                return true
            }
           

            enrollments.push(enrollment)
        });



        //assign enrollments array to Enrollments property of the enrollmentVM object
        var enrollmentVM = {};
        enrollmentVM.Enrollments = enrollments;

        //send enrollmentVM object to server
        sendArrayToServer(enrollmentVM)
    }

    function sendArrayToServer(enrolVM) {
        $.ajax({
            type: "POST",
            url: "/Enrollment/SaveEnrollment",
            data: JSON.stringify(enrolVM),
            contentType: "application/json; charset=utf-8",
            dataType: "json",
            success: function(r) {
                toastr.success(r + " records successfully saved")
            }
        });
    }
